30 void IrSender::sendIrSignal(const IrSignal& irSignal, unsigned int noSends) {
31  if (!irSignal.getIntro().isEmpty())
32  send(irSignal.getIntro(), irSignal.getFrequency());
33  for (unsigned int i = 0; i < irSignal.noRepetitions(noSends); i++)
34  send(irSignal.getRepeat(), irSignal.getFrequency());
35  if (!irSignal.getEnding().isEmpty())
36  send(irSignal.getEnding(), irSignal.getFrequency());

59 void IrSender::send(const IrSequence& irSequence, frequency_t frequency, dutycycle_t dutyCycle) {
60  enable(frequency, dutyCycle);
61 #ifdef CONSIDER_COMPUTATIONAL_DELAYS
62  uint32_t refTime = micros();
63 #endif
64  for (unsigned int i = 0U; i < irSequence.getLength(); i++) {
65 #ifdef CONSIDER_COMPUTATIONAL_DELAYS
66 #error dssdfsdfsdf
67  microseconds_t duration = irSequence.getDurations()[i];
68  refTime += duration;
69  int32_t delay = refTime - micros(); // TODO verify overflow
70  if (delay <= 0)
71  return;
72 #else
73  microseconds_t delay = irSequence.getDurations()[i];
74 #endif
75  if (i & 1)
76  sendSpace(delay);
77  else
78  sendMark(delay);
79  }
