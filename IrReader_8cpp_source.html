21void IrReader::dump(Stream &stream) const {
22 size_t count = getDataLength();
23 for (unsigned int i = 0U; i < count; i++) {
24 if (i > 0U)
25 stream.print(" ");
26 stream.print((i & 1U) ? '-' : '+');
27 stream.print(getDuration(i), DEC);
